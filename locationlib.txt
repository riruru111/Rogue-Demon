local Sense = loadstring(game:HttpGet('https://raw.githubusercontent.com/shlexware/Sirius/request/library/sense/source.lua'))()

local function updateESP()
    local musicFolder = game:GetService("ReplicatedStorage"):FindFirstChild("Music")

    if musicFolder then
        Sense.teamSettings.enemy.enabled = musicFolder:FindFirstChild("Effects") and true or false

        Sense.teamSettings.enemy.box = musicFolder:FindFirstChild("HitAir") and true or false

        Sense.teamSettings.enemy.tracer = musicFolder:FindFirstChild("BreathInOut") and true or false

        Sense.teamSettings.enemy.chams = musicFolder:FindFirstChild("DeathSpawn") and true or false

        Sense.teamSettings.enemy.name = musicFolder:FindFirstChild("Balls") and true or false

        Sense.teamSettings.enemy.healthBar = musicFolder:FindFirstChild("Damage1") and true or false
    end
end

updateESP()

local musicFolder = game:GetService("ReplicatedStorage"):WaitForChild("Music")

musicFolder.ChildAdded:Connect(function(child)
    updateESP()
end)

musicFolder.ChildRemoved:Connect(function(child)
    updateESP()
end)
